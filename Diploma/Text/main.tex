	\begin{equation}
		\label{SystemToSolve}
		\begin{cases}
			\displaystyle \frac{\partial h}{\partial t} + 
			\frac{\partial (hv_x)}{\partial x} + 
			\frac{\partial (hv_y)}{\partial y} = 0 
			\\[10pt]
			\displaystyle \frac{\partial (hv_x)}{\partial t} + 
			\frac{\partial \left(h \left(v_x^2 - B_x^2\right) + \frac{1}{2}gh^2\right)}{\partial x} +
			\frac{\partial \left(h(v_x v_y - B_x B_y)\right)}{\partial y} +
			\left( B_0 B_x - h v_y \left( f_0 + \frac{\partial f}{\partial y} y \right) \right) =
			-h \nu \left( \frac{\partial^2 v_x}{\partial x^2} + 
			\frac{\partial^2 v_x}{\partial y^2} \right) 
			\\[10pt]
			\displaystyle \frac{\partial (hv_y)}{\partial t} + 
			\frac{\partial \left(h(v_x v_y - B_x B_y)\right)}{\partial x} +
			\frac{\partial \left(h \left(v_y^2 - B_y^2\right) + \frac{1}{2}gh^2\right)}{\partial y} +
			\left( B_0 B_y + h v_x \left( f_0 + \frac{\partial f}{\partial y} y \right) \right) =
			-h \nu \left( \frac{\partial^2 v_y}{\partial x^2} + 
			\frac{\partial^2 v_y}{\partial y^2} \right)  
			\\[10pt]
			\displaystyle \frac{\partial (hB_x)}{\partial t} +
			\frac{\partial \left(h(v_y B_x - v_x B_y)\right)}{\partial y} -
			B_0 v_x = -h \nu \left( \frac{\partial^2 B_x}{\partial x^2} + 
			\frac{\partial^2 B_x}{\partial y^2} \right) 
			\\[10pt]
			\displaystyle \frac{\partial (hB_y)}{\partial t} +
			\frac{\partial \left(h(v_x B_y - v_y B_x)\right)}{\partial x} -
			B_0 v_y = -h \nu \left( \frac{\partial^2 B_y}{\partial x^2} + 
			\frac{\partial^2 B_y}{\partial y^2} \right)
		\end{cases}
	\end{equation}
	\begin{equation}
		\frac{\partial H}{\partial t} +
		\frac{\partial X}{\partial x} +
		\frac{\partial Y}{\partial y} +
		F + \nu V = 0
	\end{equation}
	\begin{equation}
		\begin{cases}
			H =
			\begin{pmatrix}
				h & h v_x & h v_y & B_x & B_y
			\end{pmatrix}^T
			\\
			X =
			\begin{pmatrix}
				\displaystyle h v_x 
				& 
				\displaystyle h \left( v_x^2 - B_x^2 \right) + \frac{1}{2} g h^2
				& 
				\displaystyle h \left( v_x v_y - B_x B_y \right)
				& 
				\displaystyle 0
				& 
				\displaystyle h \left( v_x B_y - v_y B_x \right)
			\end{pmatrix}^T
			\\
			Y =
			\begin{pmatrix}
				\displaystyle h v_y 
				& 
				\displaystyle h \left( v_x v_y - B_x B_y \right)
				& 
				\displaystyle h \left( v_y^2 - B_y^2 \right) + \frac{1}{2} g h^2
				& 
				\displaystyle h \left( v_y B_x - v_x B_y \right)
				& 
				\displaystyle 0
			\end{pmatrix}^T
			\\
			F =
			\begin{pmatrix}
				\displaystyle 0
				& 
				\displaystyle B_0 B_x - h v_y \left( f_0 + \frac{\partial f}{\partial y} y \right)
				& 
				\displaystyle B_0 B_y + h v_x \left( f_0 + \frac{\partial f}{\partial y} y \right)
				& 
				\displaystyle -B_0 v_x
				& 
				\displaystyle -B_0 v_y
			\end{pmatrix}^T
			\\
			V =
			\begin{pmatrix}
				\displaystyle 0
				& 
				\displaystyle -h \left( \frac{\partial^2 v_x}{\partial x^2} + 
				\frac{\partial^2 v_x}{\partial y^2} \right) 
				& 
				\displaystyle -h \left( \frac{\partial^2 v_y}{\partial x^2} + 
				\frac{\partial^2 v_y}{\partial y^2} \right) 
				& 
				\displaystyle -h \left( \frac{\partial^2 B_x}{\partial x^2} + 
				\frac{\partial^2 B_x}{\partial y^2} \right) 
				& 
				\displaystyle -h \left( \frac{\partial^2 B_y}{\partial x^2} + 
				\frac{\partial^2 B_y}{\partial y^2} \right)
			\end{pmatrix}^T
		\end{cases}
	\end{equation}
